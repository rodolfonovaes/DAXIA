SELECT 
	  A2_COD AS COD_FORNECEDOR,
	  A2_LOJA AS LOJA_FORNECEDOR,
	  A2_NOME AS NOME_FORNECEDOR, 
	  A2_DTAVA AS DATA_AVALIACAO, 
	  A2_DTVAL AS VALIDADE_AVALIACAO, 
	  A2_FATAVA AS FATOR_AVALIACAO,
	  A5_PRODUTO AS PRODUTO,
	  A5_NOMPROD AS NOME_PROD,
	  A5_SITU AS AS SITUACAO_FORN,
	  QEG_NIVEL AS DESCR_CLASSE,
	  QEG_IQI_I AS IQF_INFERIOR,
	  QEG_IQI_S AS IQF_SUPERIOR,
	  QEV_ANO AS ANO_REFERENCIA,
	  QEV_MES AS MES_REFERENCIA
	  FROM SA2010 SA2
	  INNER JOIN SA5010 SA5 ON A2_COD = A5_FORNECE AND A2_LOJA = A5_LOJA
	  INNER JOIN QEG010 QEG ON A5_SITU = QEG_SITU
	  INNER JOIN SB1010 SB1 ON A5_PRODUTO = B1_COD AND B1_TIPOCQ = 'Q'
	  INNER JOIN QEV010 QEV ON A5_PRODUTO = QEV_PRODUT AND QEV_FORNEC = A5_FORNECE
	  WHERE SA2.D_E_L_E_T_ = ' ' AND SA5.D_E_L_E_T_ = ' ' AND QEG.D_E_L_E_T_ = ' '
	  AND SA2.A2_COD = '000038'
