		 ZZI_MES MES,

		 ZZI_DSCPRO DESC_PRODUTO,

		 AND ZZI_DSCPRO BETWEEN ' ' AND 'ZZZZZZZ'

		 AND ZZI_EXPORT BETWEEN ' ' AND 'ZZZZZZZ'
